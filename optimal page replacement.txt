#include <iostream>
#include <vector>
#include <iomanip>
using namespace std;

int main() {
    int n, f;
    cout << "Enter the number of page references: ";
    cin >> n;

    cout << "Enter the page reference sequence: ";
    vector<int> pages(n);
    for (int i = 0; i < n; i++) cin >> pages[i];

    cout << "Enter the number of frames: ";
    cin >> f;

    vector<int> frames;                 // holds pages in frames
    int hits = 0, faults = 0;

    cout << "\n===== Optimal Page Replacement Simulation =====\n";

    for (int i = 0; i < n; i++) {
        int p = pages[i];
        bool hit = false;

        
        for (int x : frames) if (x == p) { hit = true; break; }
        if (hit) hits++;
        else {
            faults++;
            if ((int)frames.size() < f) {
                frames.push_back(p);
            } else {
                // choose victim: page used farthest in the future (or never)
                int replaceIdx = 0, farthest = -1;
                for (int j = 0; j < f; j++) {
                    int nextUse = -1;
                    for (int k = i + 1; k < n; k++) if (frames[j] == pages[k]) { nextUse = k; break; }
                    if (nextUse == -1) { replaceIdx = j; break; }     // never used again
                    if (nextUse > farthest) { farthest = nextUse; replaceIdx = j; }
                }
                frames[replaceIdx] = p;
            }
        }

        
        cout << "After referencing page " << p << " \u2192 ";
        for (int j = 0; j < f; j++) {
            if (j < (int)frames.size()) cout << frames[j] << " ";
            else cout << "- ";
        }
        cout << (hit ? " (Hit)" : " (Fault)") << "\n";
    }

    cout << "\n===== Final Results =====\n";
    cout << "Total Page References: " << n << "\n";
    cout << "Total Page Hits:       " << hits << "\n";
    cout << "Total Page Faults:     " << faults << "\n";
    cout << fixed << setprecision(4);
    cout << "Hit Ratio:   " << (hits * 100.0 / n)  << "%\n";
    cout << "Fault Ratio: " << (faults * 100.0 / n) << "%\n";
    return 0;
}
