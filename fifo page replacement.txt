#include <iostream>
#include <vector>
using namespace std;

int main() {
    int f, n;
    cout << "Enter the number of frames: ";
    cin >> f;

    cout << "Enter the number of page references: ";
    cin >> n;

    cout << "Enter the page reference sequence (space separated): \n";
    vector<int> refs(n);
    for (int i = 0; i < n; i++) cin >> refs[i];

    vector<int> frames;               // FIFO order: front = oldest
    int hits = 0, faults = 0;

    for (int x : refs) {
        bool hit = false;
        for (int v : frames) if (v == x) { hit = true; break; }

        if (hit) {
            hits++;
        } else {
            faults++;
            if ((int)frames.size() == f) frames.erase(frames.begin()); // remove oldest
            frames.push_back(x);                                       // add newest
        }

        cout << "Frames: ";
        for (int v : frames) cout << v << " ";
        cout << "\n";
    }

    cout << "\n===== FIFO Page Replacement Result =====\n";
    cout << "Total Page Hits:    " << hits << "\n";
    cout << "Total Page Faults:  " << faults << "\n";
    cout << "Hit Ratio:   " << (hits * 100.0 / n) << "%\n";
    cout << "Fault Ratio: " << (faults * 100.0 / n) << "%\n";
    return 0;
}
