#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, f;
    cout << "Enter the number of pages: ";
    cin >> n;

    vector<int> pages(n);
    cout << "Enter the reference string (page numbers): ";
    for (int i = 0; i < n; i++) cin >> pages[i];

    cout << "Enter the number of frames: ";
    cin >> f;

    vector<int> frames(f, -1);
    int pageHits = 0, pageFaults = 0;

    cout << "\n===== LRU Page Replacement Simulation =====\n";

    for (int i = 0; i < n; i++) {
        int page = pages[i];
        bool hit = false;

        
        for (int j = 0; j < f; j++) {
            if (frames[j] == page) { hit = true; pageHits++; break; }
        }

        if (!hit) {
            pageFaults++;
            int pos = -1;

            // Empty frame?
            for (int j = 0; j < f; j++) if (frames[j] == -1) { pos = j; break; }

            if (pos == -1) {
                // No empty frame: find LRU by looking back
                int bestAge = -1, idx = 0;
                for (int j = 0; j < f; j++) {
                    int age = 0;
                    for (int k = i - 1; k >= 0; k--) {
                        if (pages[k] == frames[j]) break;
                        age++;
                    }
                    if (age > bestAge) { bestAge = age; idx = j; }
                }
                pos = idx;
            }
            frames[pos] = page;
        }

        cout << "After referencing page " << page << " \xE2\x86\x92 Frames: ";
        for (int j = 0; j < f; j++) cout << (frames[j] == -1 ? "- " : to_string(frames[j]) + " ");
        cout << (hit ? " (Hit)" : " (Fault)") << "\n";
    }

    cout << "\n===== Final Results =====\n";
    cout << "Total Page References: " << n << "\n";
    cout << "Total Page Hits:       " << pageHits << "\n";
    cout << "Total Page Faults:     " << pageFaults << "\n";
    cout << "Hit Ratio:   " << ( (float)pageHits / n ) * 100 << "%\n";
    cout << "Fault Ratio: " << ( (float)pageFaults / n ) * 100 << "%\n";
    return 0;
}
